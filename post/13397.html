509 斐波那契数

斐波那契数 （通常用 F(n) 表示）形成的序列称为 斐波那契数列 。该数列由 0 和 1 开始，后面的每一项数字都是前面两项数字的和。也就是：

F(0) = 0，F(1) = 1
F(n) = F(n - 1) + F(n - 2)，其中 n > 1

给定n，请计算F(n)。

提示

0 <= n <= 30

动态规划

滚动数组

class Solution {
public:
    int fib(int n) {
        if(n<2){
            return n;
        }
        int m = 0,b=1,l=1;
        for(int i = 2;i<n;i++){
            m = b;
            b = l;
            l = m + b;
        }
        return l; 
        } 
    };

class Solution {
public:
    int fib(int n) {
        if(n<=1)return n;

        int a[2]={0,1};

        int sum = 0;
        for(int i = 2;i <= n; i++ ){
            sum = a[1] + a[0];
            a[0]=a[1];
            a[1]=sum;
        }

        return a[1];
    }
};

class Solution {
public:
    int fib(int n) {
        if(n==0) return 0;
        if(n==1) return 1;
        return fib(n-1)+fib(n-2);
    }
};
